Create `uninstall.sh`:

```sh
#!/bin/sh

set -eu

APP_NAME="agentenv"
REPO_OWNER="windoliver"
REPO_NAME="agentenv"
REPO_FULL_NAME="${AGENTENV_REPO:-${REPO_OWNER}/${REPO_NAME}}"
INSTALLER_SENTINEL="# agentenv installer"

: "${HOME:?HOME must be set}"

YES=0
DRY_RUN=0
KEEP_OPENSHELL=0
KEEP_DRIVERS=0
KEEP_CREDENTIALS=0
KEEP_DATA=0
DELETE_MODELS=0
INSTALL_DIR="${AGENTENV_INSTALL_DIR:-$HOME/.agentenv/bin}"
AGENTENV_HOME="${AGENTENV_HOME:-$HOME/.agentenv}"
AGENTENV_BIN="${AGENTENV_BIN:-$INSTALL_DIR/$APP_NAME}"
TMP_ROOT=""
PLAN_FILE=""
ACTIONS_LOG=""
ERRORS_LOG=""
FAILURE_COUNT=0

usage() {
    cat <<EOF
Usage: uninstall.sh [options]

Removes ${APP_NAME} user-level files and shell PATH entries.

Options:
  -y, --yes              Skip confirmation.
  --keep-openshell       Preserve OpenShell binary and state. This is the default.
  --keep-drivers         Preserve subprocess drivers under ~/.agentenv/drivers.
  --keep-credentials     Preserve credentials.json.
  --keep-data            Preserve env registry data under ~/.agentenv/envs.
  --delete-models        Remove agentenv-owned local model cache under ~/.agentenv/models.
  --dry-run              Print the plan without deleting anything.
  -h, --help             Show this help text.
EOF
}

log_action() {
    printf '%s\n' "$*" >> "${ACTIONS_LOG}"
}

record_error() {
    FAILURE_COUNT=$((FAILURE_COUNT + 1))
    printf '%s\n' "$*" >> "${ERRORS_LOG}"
}

parse_uninstall_args() {
    while [ "$#" -gt 0 ]; do
        case "$1" in
            -y|--yes)
                YES=1
                ;;
            --dry-run)
                DRY_RUN=1
                ;;
            --keep-openshell)
                KEEP_OPENSHELL=1
                ;;
            --keep-drivers)
                KEEP_DRIVERS=1
                ;;
            --keep-credentials)
                KEEP_CREDENTIALS=1
                ;;
            --keep-data)
                KEEP_DATA=1
                ;;
            --delete-models)
                DELETE_MODELS=1
                ;;
            -h|--help)
                usage
                exit 0
                ;;
            *)
                printf 'ERROR unknown option: %s\n' "$1" >&2
                exit 2
                ;;
        esac
        shift
    done
}

path_size_kib() {
    path=$1
    if [ ! -e "${path}" ]; then
        printf '0'
        return 0
    fi
    du -sk "${path}" 2>/dev/null | awk '{print $1}' || printf '0'
}

plan_remove_path() {
    path=$1
    if [ -e "${path}" ]; then
        size=$(path_size_kib "${path}")
        printf '  remove %s (%s KiB)\n' "${path}" "${size}" >> "${PLAN_FILE}"
    else
        printf '  already absent %s\n' "${path}" >> "${PLAN_FILE}"
    fi
}

plan_preserve_path() {
    path=$1
    reason=$2
    printf '  preserve %s (%s)\n' "${path}" "${reason}" >> "${PLAN_FILE}"
}

append_shell_plan() {
    found=0
    for rc_file in "$HOME/.bashrc" "$HOME/.zshrc" "$HOME/.profile"; do
        if [ -f "${rc_file}" ] && grep -F "${INSTALLER_SENTINEL}" "${rc_file}" >/dev/null 2>&1; then
            printf '  remove installer PATH block from %s (backup first)\n' "${rc_file}" >> "${PLAN_FILE}"
            found=1
        fi
    done
    if [ "${found}" -eq 0 ]; then
        printf '  no installer PATH blocks found\n' >> "${PLAN_FILE}"
    fi
}

build_uninstall_plan() {
    : > "${PLAN_FILE}"
    printf 'Uninstall plan\n' >> "${PLAN_FILE}"
    printf '\nRemove:\n' >> "${PLAN_FILE}"
    plan_remove_path "${AGENTENV_BIN}"
    if [ "${KEEP_DATA}" -eq 0 ]; then
        plan_remove_path "${AGENTENV_HOME}/envs"
    fi
    if [ "${KEEP_DRIVERS}" -eq 0 ]; then
        plan_remove_path "${AGENTENV_HOME}/drivers"
    fi
    if [ "${KEEP_CREDENTIALS}" -eq 0 ]; then
        plan_remove_path "${AGENTENV_HOME}/credentials.json"
    fi
    plan_remove_path "${AGENTENV_HOME}/events.db"
    plan_remove_path "${AGENTENV_HOME}/audit.key"
    plan_remove_path "${AGENTENV_HOME}/audit-signing-key"
    if [ "${DELETE_MODELS}" -eq 1 ]; then
        plan_remove_path "${AGENTENV_HOME}/models"
    fi
    printf '\nShell startup files:\n' >> "${PLAN_FILE}"
    append_shell_plan
    printf '\nPreserve:\n' >> "${PLAN_FILE}"
    plan_preserve_path "openshell" "not owned by agentenv"
    if [ "${KEEP_DATA}" -eq 1 ]; then
        plan_preserve_path "${AGENTENV_HOME}/envs" "--keep-data"
    fi
    if [ "${KEEP_DRIVERS}" -eq 1 ]; then
        plan_preserve_path "${AGENTENV_HOME}/drivers" "--keep-drivers"
    fi
    if [ "${KEEP_CREDENTIALS}" -eq 1 ]; then
        plan_preserve_path "${AGENTENV_HOME}/credentials.json" "--keep-credentials"
    fi
}

confirm_uninstall() {
    if [ "${YES}" -eq 1 ]; then
        return 0
    fi
    if [ ! -r /dev/tty ]; then
        printf 'ERROR confirmation required; rerun with --yes to uninstall non-interactively\n' >&2
        return 1
    fi
    printf 'Proceed with agentenv uninstall? [y/N] ' > /dev/tty
    IFS= read -r answer < /dev/tty || return 1
    case "${answer}" in
        y|Y|yes|YES) return 0 ;;
        *) return 1 ;;
    esac
}

cleanup_uninstall_tmp() {
    if [ -n "${TMP_ROOT}" ] && [ -d "${TMP_ROOT}" ] && [ "${FAILURE_COUNT}" -eq 0 ]; then
        rm -rf "${TMP_ROOT}"
    fi
}

main() {
    parse_uninstall_args "$@"
    TMP_ROOT=$(mktemp -d "${TMPDIR:-/tmp}/${APP_NAME}-uninstall.XXXXXX")
    PLAN_FILE="${TMP_ROOT}/plan.txt"
    ACTIONS_LOG="${TMP_ROOT}/actions.log"
    ERRORS_LOG="${TMP_ROOT}/errors.log"
    : > "${ACTIONS_LOG}"
    : > "${ERRORS_LOG}"
    trap cleanup_uninstall_tmp EXIT INT TERM

    build_uninstall_plan
    cat "${PLAN_FILE}"
    if [ "${DRY_RUN}" -eq 1 ]; then
        return 0
    fi
    confirm_uninstall
}

if [ "${AGENTENV_UNINSTALLER_SOURCE_ONLY:-0}" != "1" ]; then
    main "$@"
fi
```

```sh
backup_uninstall_file() {
    file_path=$1
    [ -f "${file_path}" ] || return 0
    timestamp=$(date -u '+%Y%m%d%H%M%S')
    cp "${file_path}" "${file_path}.agentenv.bak.${timestamp}"
}

remove_installer_block() {
    rc_file=$1
    [ -f "${rc_file}" ] || return 0
    if ! grep -F "${INSTALLER_SENTINEL}" "${rc_file}" >/dev/null 2>&1; then
        return 0
    fi

    backup_uninstall_file "${rc_file}" || {
        record_error "failed to back up ${rc_file}"
        return 0
    }
    tmp_file="${TMP_ROOT}/$(basename "${rc_file}").cleaned"
    awk -v sentinel="${INSTALLER_SENTINEL}" '
        $0 == sentinel {
            getline
            if ($0 ~ /^export PATH=/) {
                next
            }
            print
            next
        }
        { print }
    ' "${rc_file}" > "${tmp_file}" || {
        record_error "failed to clean installer PATH block from ${rc_file}"
        return 0
    }
    mv "${tmp_file}" "${rc_file}" || {
        record_error "failed to replace ${rc_file}"
        return 0
    }
    log_action "removed installer PATH block from ${rc_file}"
}

remove_path_if_present() {
    path=$1
    label=$2
    if [ ! -e "${path}" ]; then
        log_action "already absent ${label}: ${path}"
        return 0
    fi
    if rm -rf "${path}"; then
        log_action "removed ${label}: ${path}"
    else
        record_error "failed to remove ${label}: ${path}"
    fi
}

remove_shell_path_blocks() {
    remove_installer_block "$HOME/.bashrc"
    remove_installer_block "$HOME/.zshrc"
    remove_installer_block "$HOME/.profile"
}

remove_selected_paths() {
    remove_path_if_present "${AGENTENV_BIN}" "agentenv binary"
    if [ "${KEEP_DATA}" -eq 0 ]; then
        remove_path_if_present "${AGENTENV_HOME}/envs" "env data"
    fi
    if [ "${KEEP_DRIVERS}" -eq 0 ]; then
        remove_path_if_present "${AGENTENV_HOME}/drivers" "drivers"
    fi
    if [ "${KEEP_CREDENTIALS}" -eq 0 ]; then
        remove_path_if_present "${AGENTENV_HOME}/credentials.json" "credentials json"
    fi
    remove_path_if_present "${AGENTENV_HOME}/events.db" "global events db"
    remove_path_if_present "${AGENTENV_HOME}/audit.key" "audit key"
    remove_path_if_present "${AGENTENV_HOME}/audit-signing-key" "audit signing key"
    if [ "${DELETE_MODELS}" -eq 1 ]; then
        remove_path_if_present "${AGENTENV_HOME}/models" "agentenv model cache"
    fi
    rmdir "${INSTALL_DIR}" 2>/dev/null || true
    rmdir "${AGENTENV_HOME}" 2>/dev/null || true
}

execute_uninstall() {
```

```sh
            log_action "destroyed env ${env_name}"

    done
```

```rust
fn run_uninstall(args: UninstallArgs) -> Result<()> {
    let script = resolve_uninstall_script().context("resolve uninstall script")?;
    let mut command = process::Command::new("sh");
    command.arg(&script);
    for arg in args.to_script_args() {
        command.arg(arg);
    }
    let status = command
        .status()
        .with_context(|| format!("run uninstall script at {}", script.display()))?;
    match status.code() {
        Some(0) => Ok(()),
        Some(code) => exit_process(code),
        None => bail!("uninstall script terminated by signal"),
    }
}

fn resolve_uninstall_script() -> Result<PathBuf> {
    if let Some(path) = std::env::var_os("AGENTENV_UNINSTALL_SCRIPT") {
        let path = PathBuf::from(path);
        if path.is_file() {
            return Ok(path);
        }
        bail!("AGENTENV_UNINSTALL_SCRIPT does not point to a file: {}", path.display());
    }

    if let Ok(current_exe) = std::env::current_exe() {
        if let Some(parent) = current_exe.parent() {
            let sibling = parent.join("uninstall.sh");
            if sibling.is_file() {
                return Ok(sibling);
            }
        }
    }

    let cwd = std::env::current_dir().context("determine current directory")?;
    if let Some(repo_script) = find_uninstall_script_in_ancestors(&cwd) {
        return Ok(repo_script);
    }

    download_hosted_uninstall_script()
}

fn find_uninstall_script_in_ancestors(start: &Path) -> Option<PathBuf> {
    for dir in start.ancestors() {
        let candidate = dir.join("uninstall.sh");
        if candidate.is_file() {
            return Some(candidate);
        }
    }
    None
}
```

```rust
fn download_hosted_uninstall_script() -> Result<PathBuf> {
    let repo = std::env::var("AGENTENV_REPO").unwrap_or_else(|_| "windoliver/agentenv".to_owned());
    let release_base = std::env::var("AGENTENV_RELEASE_BASE_URL")
        .unwrap_or_else(|_| format!("https://github.com/{repo}/releases/download"));
    let version = std::env::var("AGENTENV_VERSION")
        .ok()
        .filter(|value| !value.trim().is_empty())
        .unwrap_or_else(|| format!("v{}", env!("CARGO_PKG_VERSION")));
    let base = format!(
        "{}/{}/",
        release_base.trim_end_matches('/'),
        version.trim_start_matches("refs/tags/")
    );
    let tmp_dir = make_uninstall_download_dir()?;
    let script_path = tmp_dir.join("uninstall.sh");
    let checksum_path = tmp_dir.join("uninstall.sh.sha256");
    download_url_to_file(&format!("{base}uninstall.sh"), &script_path)?;
    download_url_to_file(&format!("{base}uninstall.sh.sha256"), &checksum_path)?;
    verify_file_sha256(&script_path, &checksum_path)?;
    Ok(script_path)
}

fn make_uninstall_download_dir() -> Result<PathBuf> {
    let nanos = SystemTime::now()
        .duration_since(UNIX_EPOCH)
        .context("system clock before unix epoch")?
        .as_nanos();
    let dir = std::env::temp_dir().join(format!("agentenv-uninstall-{}-{nanos}", process::id()));
    fs::create_dir_all(&dir)
        .with_context(|| format!("create uninstall download directory {}", dir.display()))?;
    Ok(dir)
}

fn download_url_to_file(url: &str, path: &Path) -> Result<()> {
    let curl_status = process::Command::new("curl")
        .arg("-fLsS")
        .arg(url)
        .arg("-o")
        .arg(path)
        .status();
    if let Ok(status) = curl_status {
        if status.success() {
            return Ok(());
        }
    }

    let status = process::Command::new("wget")
        .arg("-qO")
        .arg(path)
        .arg(url)
        .status()
        .with_context(|| format!("download {url} with curl or wget"))?;
    if status.success() {
        Ok(())
    } else {
        bail!("download {url} failed with status {status}");
    }
}

fn verify_file_sha256(file_path: &Path, checksum_path: &Path) -> Result<()> {
    let expected_text = fs::read_to_string(checksum_path)
        .with_context(|| format!("read checksum {}", checksum_path.display()))?;
    let expected = expected_text
        .split_whitespace()
        .next()
        .context("checksum file is empty")?
        .to_ascii_lowercase();
    let bytes = fs::read(file_path).with_context(|| format!("read {}", file_path.display()))?;
    let actual = hex::encode(Sha256::digest(&bytes));
    if actual == expected {
        Ok(())
    } else {
        bail!(
            "checksum verification failed for {}: expected {expected}, got {actual}",
            file_path.display()
        );
    }
}
```
