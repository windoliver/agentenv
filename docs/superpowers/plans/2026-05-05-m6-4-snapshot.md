use ed25519_dalek::{Signature, Signer, SigningKey, Verifier, VerifyingKey};
use rand_core::{OsRng, RngCore};
use serde::{Deserialize, Serialize};
use sha2::{Digest, Sha256};
use thiserror::Error;

pub const SNAPSHOT_VERSION: &str = "0.1.0";

pub type SnapshotResult<T> = Result<T, SnapshotError>;

#[derive(Debug, Error)]
pub enum SnapshotError {
    #[error("snapshot IO error at `{path}`: {source}")]
    Io {
        path: PathBuf,
        source: std::io::Error,
    },
    #[error("snapshot JSON error at `{path}`: {source}")]
    Json {
        path: PathBuf,
        source: serde_json::Error,
    },
    #[error("snapshot YAML error at `{path}`: {source}")]
    Yaml {
        path: PathBuf,
        source: serde_yaml::Error,
    },
    #[error("snapshot hex error: {0}")]
    Hex(#[from] hex::FromHexError),
    #[error("snapshot signature error")]
    Signature(#[from] ed25519_dalek::SignatureError),
    #[error("snapshot path `{path}` is invalid: {message}")]
    InvalidPath { path: String, message: String },
    #[error("snapshot digest mismatch for `{path}`")]
    DigestMismatch { path: String },
    #[error("snapshot merkle root mismatch")]
    MerkleRootMismatch,
    #[error("snapshot manifest hash mismatch")]
    ManifestHashMismatch,
    #[error("snapshot signature verification failed")]
    SignatureVerification,
    #[error("snapshot payload `{path}` is not listed in manifest")]
    ExtraPayload { path: String },
    #[error("snapshot requires a workspace payload")]
    MissingWorkspace,
    #[error("snapshot schema version `{version}` is unsupported")]
    UnsupportedVersion { version: String },
    #[error("snapshot signing key must contain exactly 32 raw bytes")]
    InvalidSigningKeyLength,
}

#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
#[serde(rename_all = "snake_case", deny_unknown_fields)]
pub enum SnapshotEntryKind {
    File,
    Directory,
    Symlink,
}

#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
#[serde(deny_unknown_fields)]
pub struct SnapshotFileEntry {
    pub path: String,
    pub kind: SnapshotEntryKind,
    pub mode: String,
    pub size: u64,
    pub sha256: String,
}

#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
#[serde(deny_unknown_fields)]
pub struct SnapshotSection {
    pub path: String,
    pub sha256: String,
    #[serde(default, skip_serializing_if = "Option::is_none")]
    pub kind: Option<SnapshotEntryKind>,
}

#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
#[serde(deny_unknown_fields)]
pub struct SnapshotCredentialRequirement {
    pub name: String,
    pub source: String,
    #[serde(default, skip_serializing_if = "Option::is_none")]
    pub reference: Option<String>,
    #[serde(default, skip_serializing_if = "Option::is_none")]
    pub required: Option<bool>,
}

#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
#[serde(deny_unknown_fields)]
pub struct SnapshotStrippedEntry {
    pub path: String,
    pub reason: String,
}

#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
#[serde(deny_unknown_fields)]
pub struct SnapshotManifest {
    pub version: String,
    pub agentenv_version: String,
    pub source_env: String,
    pub created_at: String,
    pub min_agentenv_version: String,
    pub driver_protocol_version: String,
    pub sections: BTreeMap<String, SnapshotSection>,
    pub files: Vec<SnapshotFileEntry>,
    #[serde(default, skip_serializing_if = "Vec::is_empty")]
    pub credential_requirements: Vec<SnapshotCredentialRequirement>,
    #[serde(default, skip_serializing_if = "Vec::is_empty")]
    pub stripped: Vec<SnapshotStrippedEntry>,
    pub merkle_root: String,
}

#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
#[serde(deny_unknown_fields)]
pub struct SnapshotSignatures {
    pub version: String,
    pub signature_algorithm: String,
    pub hash_algorithm: String,
    pub public_key: String,
    pub manifest_sha256: String,
    pub merkle_root: String,
    pub signature: String,
}

#[derive(Debug, Clone, PartialEq, Eq)]
struct Inventory {
    files: Vec<SnapshotFileEntry>,
    merkle_root: String,
    sections: BTreeMap<String, SnapshotSection>,
}

pub fn write_manifest_and_signature(snapshot_dir: &Path, key_path: &Path) -> SnapshotResult<()> {
    let inventory = build_inventory(snapshot_dir)?;
    if !snapshot_dir.join("workspace").is_dir() {
        return Err(SnapshotError::MissingWorkspace);
    }
    let manifest = SnapshotManifest {
        version: SNAPSHOT_VERSION.to_owned(),
        agentenv_version: env!("CARGO_PKG_VERSION").to_owned(),
        source_env: "test-env".to_owned(),
        created_at: "2026-05-05T00:00:00Z".to_owned(),
        min_agentenv_version: env!("CARGO_PKG_VERSION").to_owned(),
        driver_protocol_version: agentenv_proto::SCHEMA_VERSION.to_owned(),
        sections: inventory.sections,
        files: inventory.files,
        credential_requirements: Vec::new(),
        stripped: Vec::new(),
        merkle_root: inventory.merkle_root,
    };
    write_signed_manifest(snapshot_dir, key_path, &manifest)
}

pub fn write_signed_manifest(
    snapshot_dir: &Path,
    key_path: &Path,
    manifest: &SnapshotManifest,
) -> SnapshotResult<()> {
    let manifest_path = snapshot_dir.join("manifest.json");
    write_json(&manifest_path, manifest)?;
    let manifest_sha256 = digest_file_prefixed(&manifest_path)?;
    let key = load_or_create_signing_key(key_path)?;
    let message = signed_message(&manifest_sha256, &manifest.merkle_root)?;
    let signature = key.sign(&message);
    let signatures = SnapshotSignatures {
        version: SNAPSHOT_VERSION.to_owned(),
        signature_algorithm: "ed25519".to_owned(),
        hash_algorithm: "sha256".to_owned(),
        public_key: hex::encode(key.verifying_key().to_bytes()),
        manifest_sha256,
        merkle_root: manifest.merkle_root.clone(),
        signature: hex::encode(signature.to_bytes()),
    };
    write_json(&snapshot_dir.join("signatures.json"), &signatures)
}

pub fn verify_snapshot_dir(snapshot_dir: &Path) -> SnapshotResult<SnapshotManifest> {
    let manifest_path = snapshot_dir.join("manifest.json");
    let signatures_path = snapshot_dir.join("signatures.json");
    let manifest: SnapshotManifest = read_json(&manifest_path)?;
    let signatures = read_signatures(&signatures_path)?;
    if manifest.version != SNAPSHOT_VERSION {
        return Err(SnapshotError::UnsupportedVersion {
            version: manifest.version,
        });
    }

    let actual = build_inventory(snapshot_dir)?;
    let expected_paths = manifest
        .files
        .iter()
        .map(|entry| entry.path.as_str())
        .collect::<std::collections::BTreeSet<_>>();
    for entry in &actual.files {
        if !expected_paths.contains(entry.path.as_str()) {
            return Err(SnapshotError::ExtraPayload {
                path: entry.path.clone(),
            });
        }
    }
    for expected in &manifest.files {
        validate_relative_path(&expected.path)?;
        let Some(actual_entry) = actual.files.iter().find(|entry| entry.path == expected.path)
        else {
            return Err(SnapshotError::DigestMismatch {
                path: expected.path.clone(),
            });
        };
        if actual_entry.sha256 != expected.sha256 {
            return Err(SnapshotError::DigestMismatch {
                path: expected.path.clone(),
            });
        }
    }
    if actual.merkle_root != manifest.merkle_root || manifest.merkle_root != signatures.merkle_root
    {
        return Err(SnapshotError::MerkleRootMismatch);
    }
    let manifest_sha256 = digest_file_prefixed(&manifest_path)?;
    if manifest_sha256 != signatures.manifest_sha256 {
        return Err(SnapshotError::ManifestHashMismatch);
    }
    verify_signature(&signatures, &manifest_sha256, &manifest.merkle_root)?;
    Ok(manifest)
}

pub fn read_signatures(path: &Path) -> SnapshotResult<SnapshotSignatures> {
    read_json(path)
}

fn build_inventory(root: &Path) -> SnapshotResult<Inventory> {
    let mut files = Vec::new();
    collect_entries(root, root, &mut files)?;
    files.retain(|entry| entry.path != "manifest.json" && entry.path != "signatures.json");
    files.sort_by(|left, right| left.path.cmp(&right.path));
    let merkle_root = digest_inventory(&files);
    let mut sections = BTreeMap::new();
    for section in ["blueprint.yaml", "lock.yaml", "policy.yaml", "events.db"] {
        if let Some(entry) = files.iter().find(|entry| entry.path == section) {
            sections.insert(
                section.trim_end_matches(".yaml").trim_end_matches(".db").to_owned(),
                SnapshotSection {
                    path: section.to_owned(),
                    sha256: entry.sha256.clone(),
                    kind: Some(entry.kind.clone()),
                },
            );
        }
    }
    for section in ["workspace", "home"] {
        let prefix = format!("{section}/");
        let digest = digest_section(&files, &prefix);
        if let Some(sha256) = digest {
            sections.insert(
                section.to_owned(),
                SnapshotSection {
                    path: section.to_owned(),
                    sha256,
                    kind: Some(SnapshotEntryKind::Directory),
                },
            );
        }
    }
    Ok(Inventory {
        files,
        merkle_root,
        sections,
    })
}

fn collect_entries(root: &Path, dir: &Path, entries: &mut Vec<SnapshotFileEntry>) -> SnapshotResult<()> {
    for child in read_dir_sorted(dir)? {
        let relative = relative_path(root, &child)?;
        validate_relative_path(&relative)?;
        let metadata = fs::symlink_metadata(&child).map_err(|source| SnapshotError::Io {
            path: child.clone(),
            source,
        })?;
        if metadata.is_dir() {
            collect_entries(root, &child, entries)?;
        } else if metadata.file_type().is_symlink() {
            let target = fs::read_link(&child).map_err(|source| SnapshotError::Io {
                path: child.clone(),
                source,
            })?;
            entries.push(SnapshotFileEntry {
                path: relative,
                kind: SnapshotEntryKind::Symlink,
                mode: "symlink".to_owned(),
                size: target.as_os_str().as_encoded_bytes().len() as u64,
                sha256: digest_bytes_prefixed(target.as_os_str().as_encoded_bytes()),
            });
        } else if metadata.is_file() {
            let bytes = fs::read(&child).map_err(|source| SnapshotError::Io {
                path: child.clone(),
                source,
            })?;
            entries.push(SnapshotFileEntry {
                path: relative,
                kind: SnapshotEntryKind::File,
                mode: "file".to_owned(),
                size: bytes.len() as u64,
                sha256: digest_bytes_prefixed(&bytes),
            });
        }
    }
    Ok(())
}

fn read_dir_sorted(dir: &Path) -> SnapshotResult<Vec<PathBuf>> {
    let mut entries = fs::read_dir(dir)
        .map_err(|source| SnapshotError::Io {
            path: dir.to_path_buf(),
            source,
        })?
        .map(|entry| {
            entry
                .map(|entry| entry.path())
                .map_err(|source| SnapshotError::Io {
                    path: dir.to_path_buf(),
                    source,
                })
        })
        .collect::<SnapshotResult<Vec<_>>>()?;
    entries.sort();
    Ok(entries)
}

fn digest_inventory(files: &[SnapshotFileEntry]) -> String {
    let mut hasher = Sha256::new();
    for entry in files {
        hasher.update(entry.path.as_bytes());
        hasher.update([0]);
        hasher.update(format!("{:?}", entry.kind).as_bytes());
        hasher.update([0]);
        hasher.update(entry.sha256.as_bytes());
        hasher.update([0]);
    }
    format!("sha256:{}", hex::encode(hasher.finalize()))
}

fn digest_section(files: &[SnapshotFileEntry], prefix: &str) -> Option<String> {
    let section_files = files
        .iter()
        .filter(|entry| entry.path.starts_with(prefix))
        .cloned()
        .collect::<Vec<_>>();
    (!section_files.is_empty()).then(|| digest_inventory(&section_files))
}

fn digest_file_prefixed(path: &Path) -> SnapshotResult<String> {
    let bytes = fs::read(path).map_err(|source| SnapshotError::Io {
        path: path.to_path_buf(),
        source,
    })?;
    Ok(digest_bytes_prefixed(&bytes))
}

fn digest_bytes_prefixed(bytes: &[u8]) -> String {
    let mut hasher = Sha256::new();
    hasher.update(bytes);
    format!("sha256:{}", hex::encode(hasher.finalize()))
}

fn write_json<T: Serialize>(path: &Path, value: &T) -> SnapshotResult<()> {
    let rendered = serde_json::to_string_pretty(value).map_err(|source| SnapshotError::Json {
        path: path.to_path_buf(),
        source,
    })?;
    fs::write(path, rendered).map_err(|source| SnapshotError::Io {
        path: path.to_path_buf(),
        source,
    })
}

fn read_json<T: for<'de> Deserialize<'de>>(path: &Path) -> SnapshotResult<T> {
    let bytes = fs::read(path).map_err(|source| SnapshotError::Io {
        path: path.to_path_buf(),
        source,
    })?;
    serde_json::from_slice(&bytes).map_err(|source| SnapshotError::Json {
        path: path.to_path_buf(),
        source,
    })
}

fn load_or_create_signing_key(path: &Path) -> SnapshotResult<SigningKey> {
    if path.exists() {
        let bytes = fs::read(path).map_err(|source| SnapshotError::Io {
            path: path.to_path_buf(),
            source,
        })?;
        let secret: [u8; 32] = bytes
            .try_into()
            .map_err(|_| SnapshotError::InvalidSigningKeyLength)?;
        return Ok(SigningKey::from_bytes(&secret));
    }
    let mut secret = [0u8; 32];
    OsRng.fill_bytes(&mut secret);
    fs::write(path, secret).map_err(|source| SnapshotError::Io {
        path: path.to_path_buf(),
        source,
    })?;
    Ok(SigningKey::from_bytes(&secret))
}

fn verify_signature(
    signatures: &SnapshotSignatures,
    manifest_sha256: &str,
    merkle_root: &str,
) -> SnapshotResult<()> {
    let public_key_bytes: [u8; 32] = hex::decode(&signatures.public_key)?
        .try_into()
        .map_err(|_| SnapshotError::SignatureVerification)?;
    let signature_bytes: [u8; 64] = hex::decode(&signatures.signature)?
        .try_into()
        .map_err(|_| SnapshotError::SignatureVerification)?;
    let public_key = VerifyingKey::from_bytes(&public_key_bytes)?;
    let signature = Signature::from_bytes(&signature_bytes);
    let message = signed_message(manifest_sha256, merkle_root)?;
    public_key
        .verify(&message, &signature)
        .map_err(|_| SnapshotError::SignatureVerification)
}

fn signed_message(manifest_sha256: &str, merkle_root: &str) -> SnapshotResult<Vec<u8>> {
    serde_json::to_vec(&serde_json::json!({
        "manifest_sha256": manifest_sha256,
        "merkle_root": merkle_root
    }))
    .map_err(|source| SnapshotError::Json {
        path: PathBuf::from("<signed-message>"),
        source,
    })
}

fn relative_path(root: &Path, path: &Path) -> SnapshotResult<String> {
    let relative = path.strip_prefix(root).map_err(|_| SnapshotError::InvalidPath {
        path: path.display().to_string(),
        message: "entry is outside snapshot root".to_owned(),
    })?;
    Ok(relative
        .components()
        .map(|component| component.as_os_str().to_string_lossy())
        .collect::<Vec<_>>()
        .join("/"))
}

fn validate_relative_path(path: &str) -> SnapshotResult<()> {
    if path.is_empty() || path.starts_with('/') || path.split('/').any(|part| part == "..") {
        return Err(SnapshotError::InvalidPath {
            path: path.to_owned(),
            message: "path must be relative and cannot contain `..`".to_owned(),
        });
    }
    Ok(())
}

fn sanitize_dir(root: &Path, dir: &Path, report: &mut SanitizerReport) -> SnapshotResult<()> {
    for child in read_dir_sorted(dir)? {
        let relative = relative_path(root, &child)?;
        if is_credential_path(&relative) {
            remove_path(&child)?;
            report.stripped.push(SnapshotStrippedEntry {
                path: relative,
                reason: "credential_path".to_owned(),
            });
            continue;
        }
        let metadata = fs::symlink_metadata(&child).map_err(|source| SnapshotError::Io {
            path: child.clone(),
            source,
        })?;
        if metadata.is_dir() {
            sanitize_dir(root, &child, report)?;
        } else if metadata.is_file() {
            redact_file_if_structured(&child)?;
        }
    }
    Ok(())
}

fn is_credential_path(path: &str) -> bool {
    let lower = path.to_ascii_lowercase();
    let parts = lower.split('/').collect::<Vec<_>>();
    if lower.ends_with(".config/gh/hosts.yml") {
        return true;
    }
    if parts
        .windows(2)
        .any(|pair| matches!(pair, [".agentenv" | ".codex" | ".claude" | ".openclaw", file] if file.starts_with("credentials")))
    {
        return true;
    }
    if parts
        .windows(2)
        .any(|pair| matches!(pair, [".aws", file] if file.contains("credentials")))
    {
        return true;
    }
    parts
        .last()
        .is_some_and(|basename| basename.starts_with("credentials"))
}

fn remove_path(path: &Path) -> SnapshotResult<()> {
    let metadata = fs::symlink_metadata(path).map_err(|source| SnapshotError::Io {
        path: path.to_path_buf(),

        })
    } else {
        fs::remove_file(path).map_err(|source| SnapshotError::Io {
            path: path.to_path_buf(),

fn redact_file_if_structured(path: &Path) -> SnapshotResult<()> {
    let Some(extension) = path.extension().and_then(|value| value.to_str()) else {
        return Ok(());
    };
    match extension {
        "json" => {
            let raw = fs::read_to_string(path).map_err(|source| SnapshotError::Io {
                path: path.to_path_buf(),
                source,
            })?;
            let Ok(value) = serde_json::from_str::<serde_json::Value>(&raw) else {
                return Ok(());
            };
            let redacted = redact_json(value);
            let rendered = serde_json::to_string_pretty(&redacted).map_err(|source| {
                SnapshotError::Json {
                    path: path.to_path_buf(),
                    source,
                }
            })?;
            fs::write(path, rendered).map_err(|source| SnapshotError::Io {
                path: path.to_path_buf(),
                source,
            })
        }
        "yaml" | "yml" => {
            let raw = fs::read_to_string(path).map_err(|source| SnapshotError::Io {
                path: path.to_path_buf(),
                source,
            })?;
            let Ok(value) = serde_yaml::from_str::<serde_yaml::Value>(&raw) else {
                return Ok(());
            };
            let redacted = redact_yaml(value);
            let rendered = serde_yaml::to_string(&redacted).map_err(|source| {
                SnapshotError::Yaml {
                    path: path.to_path_buf(),
                    source,
                }
            })?;
            fs::write(path, rendered).map_err(|source| SnapshotError::Io {
                path: path.to_path_buf(),
                source,
            })
        }
        _ => Ok(()),
    }
}

fn redact_json(value: serde_json::Value) -> serde_json::Value {
    match value {
        serde_json::Value::Object(map) => serde_json::Value::Object(
            map.into_iter()
                .map(|(key, value)| {
                    if is_secret_key(&key) {
                        (key, serde_json::Value::String("[redacted]".to_owned()))
                    } else {
                        (key, redact_json(value))
                    }
                })
                .collect(),
        ),
        serde_json::Value::Array(values) => {
            serde_json::Value::Array(values.into_iter().map(redact_json).collect())
        }
        other => other,
    }
}

fn redact_yaml(value: serde_yaml::Value) -> serde_yaml::Value {
    match value {
        serde_yaml::Value::Mapping(map) => serde_yaml::Value::Mapping(
            map.into_iter()
                .map(|(key, value)| {
                    let secret_key = key
                        .as_str()
                        .is_some_and(is_secret_key);
                    if secret_key {
                        (key, serde_yaml::Value::String("[redacted]".to_owned()))
                    } else {
                        (key, redact_yaml(value))
                    }
                })
                .collect(),
        ),
        serde_yaml::Value::Sequence(values) => {
            serde_yaml::Value::Sequence(values.into_iter().map(redact_yaml).collect())
        }
        other => other,
    }
}

fn is_secret_key(key: &str) -> bool {
    let lower = key.to_ascii_lowercase();
    [
        "token",
        "secret",
        "password",
        "api_key",
        "apikey",
        "authorization",
        "credential",
        "webhook",
        "mcp",
        "nexus",
    ]
    .iter()
    .any(|marker| lower.contains(marker))
}

fn scan_tree_for_secrets(root: &Path) -> SnapshotResult<()> {
    for child in read_dir_sorted(root)? {
        scan_path_for_secrets(root, &child)?;
    }
    Ok(())
}

fn scan_path_for_secrets(root: &Path, path: &Path) -> SnapshotResult<()> {
    let metadata = fs::symlink_metadata(path).map_err(|source| SnapshotError::Io {
        path: path.to_path_buf(),
        source,
    })?;
    if metadata.is_dir() {
        for child in read_dir_sorted(path)? {
            scan_path_for_secrets(root, &child)?;
        }
        return Ok(());
    }
    if !metadata.is_file() || metadata.len() > 1024 * 1024 {
        return Ok(());
    }
    let bytes = fs::read(path).map_err(|source| SnapshotError::Io {
        path: path.to_path_buf(),
        source,
    })?;
    let Ok(text) = std::str::from_utf8(&bytes) else {
        return Ok(());
    };
    if let Some(finding) = find_secret(text, &relative_path(root, path)?) {
        return Err(SnapshotError::leaked_secret(finding.path, finding.class));
    }
    Ok(())
}

fn find_secret(text: &str, path: &str) -> Option<SecretFinding> {
    let classes = [
        ("openai_api_key", "sk-"),
        ("anthropic_api_key", "sk-ant-"),
        ("github_token", "ghp_"),
        ("github_token", "gho_"),
        ("github_token", "ghu_"),
        ("github_token", "ghs_"),
        ("github_token", "ghr_"),
        ("github_token", "github_pat_"),
        ("aws_access_key", "AKIA"),
        ("aws_session_key", "ASIA"),
        ("slack_token", "xoxb-"),
        ("slack_token", "xoxp-"),
        ("slack_token", "xapp-"),
    ];
    classes
        .iter()
        .find(|(_, needle)| text.contains(needle))
        .map(|(class, _)| SecretFinding {
            path: path.to_owned(),
            class,
        })
        .or_else(|| {
            text.lines()
                .find(|line| {
                    let lower = line.to_ascii_lowercase();
                    lower.contains("webhook") && lower.contains("secret") && lower.contains(':')
                })
                .map(|_| SecretFinding {
                    path: path.to_owned(),
                    class: "webhook_secret",
                })
        })
}

fn check_snapshot_credentials(
    credentials: &mut dyn CredentialProvider,
    manifest: &crate::snapshot::SnapshotManifest,
) -> RuntimeResult<()> {
    for requirement in &manifest.credential_requirements {
        if requirement.required == Some(false) {
            continue;
        }
        let reference = requirement.reference.as_deref().unwrap_or(&requirement.name);
        match requirement.source.as_str() {
            "env" if std::env::var_os(reference).is_none() => {
                return Err(RuntimeError::MissingCredential {
                    name: reference.to_owned(),
                });
            }
            "credstore" => match credentials.backend_name(reference)? {
                Some(backend) if is_credstore_backend(&backend) => {}
                _ => {
                    return Err(RuntimeError::MissingCredential {
                        name: reference.to_owned(),
                    });
                }
            },
            _ => {}
        }
    }
    Ok(())
}

## Snapshots

`agentenv snapshot <env> [--output <dir.agentenvsnap>]` captures a
